## Using with label

You can use any node element for `label` property. For more info, refer to Label component.

<Canvas>
  <Story name="Using with label">
    {
      () => {
        const [checkboxGroupValues, setValues] = React.useState([]);
        const handleChange = (event, [value]) => {
          console.log(value);
          checkboxGroupValues.push(value);
          setValues(checkboxGroupValues);
        };
        return (
          <>
            <Label>Lorem ipsum dolor sit amet consectetur:</Label>
            <CheckboxGroup
              control="checkbox-1"
              name="checkbox-group-2"
              checkboxGroupValues={checkboxGroupValues}
              onChange={handleChange}
            >
              <Checkbox
                label="Checkbox 1"
                value="checkbox-1"
                controller="checkbox-1"
              />
              <Checkbox
                label="Checkbox 2"
                value="checkbox-2"
                controller="checkbox-1"
              />
            </CheckboxGroup>
          </>
        );
      }
    }
  </Story>
</Canvas>
