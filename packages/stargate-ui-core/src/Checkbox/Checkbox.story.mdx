## Using with CheckboxGroup

`CheckboxGroup` provides data management for `Checkbox`, can controls it and receives value when it has changes as well.

<Canvas>
  <Story name="Using with CheckboxGroup">
    {
      () => {
        const [values, setValues] = React.useState(['checkbox-1', 'checkbox-2']);
        const handleChange = (event, [, values]) => {
          setValues(values);
        };
        return (
          <>
            <Label>Lorem ipsum dolor sit amet consectetur:</Label>
            <CheckboxGroup
              control="checkbox-1"
              name="checkbox-group-2"
              value={values}
              onChange={handleChange}
            >
              <Checkbox
                label="Checkbox 1"
                value="checkbox-1"
                controller="checkbox-group-2"
              />
              <Checkbox
                label="Checkbox 2"
                value="checkbox-2"
                controller="checkbox-group-2"
              />
              <Checkbox
                label="Checkbox 3"
                value="checkbox-3"
                controller="checkbox-group-2"
              />
            </CheckboxGroup>
            <Typography>
              Values checked:
            </Typography>
            <ul>
              {
                values.map((value) => (
                  <Typography element="li">
                    {value}
                  </Typography>
                ))
              }
              {
                !values.length && (
                  <Typography element="li">
                    Empty
                  </Typography>
                )
              }
            </ul>
          </>
        );
      }
    }
  </Story>
</Canvas>
